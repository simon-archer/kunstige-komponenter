How to Use

To use this enhanced bundle in your own project, simply import it:

<!-- Import the enhanced bundle -->
<script type="module">
  import KI_COMPONENTS from 'https://cdn.jsdelivr.net/gh/simon-archer/kunstige-komponenter@master/bundle.js';
  
  // Wait for components to be ready before using them
  KI_COMPONENTS.ready(() => {
    console.log(`Loaded ${KI_COMPONENTS.name} v${KI_COMPONENTS.version}`);
    console.log(`Available icons: ${KI_COMPONENTS.svgIcons.join(', ')}`);
    
    // Now all components are guaranteed to be defined and ready
    const messageField = document.querySelector('ki-message-field');
    if (messageField) {
      messageField.addEventListener('message-sent', (e) => {
        console.log('Message sent:', e.detail.value);
      });
    }
  });
</script>

<!-- Then use any component -->
<ki-message-field placeholder="Type a message..."></ki-message-field>
